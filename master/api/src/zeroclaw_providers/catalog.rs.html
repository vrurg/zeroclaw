1//! Per-family catalog source table.
2//!
3//! Reaches the model catalog for any provider family without constructing
4//! a live `ModelProvider` (which would require typed runtime context like
5//! Azure's `resource`/`deployment` or Bedrock's `region`). Used by the
6//! gateway's `/api/config/catalog/models` endpoint and the TUI's
7//! config flow when the operator hasn't supplied a credential yet.
8//!
9//! Each family maps to a tuple `(models_dev_key, openrouter_vendor_prefix)`;
10//! `list_models_for_family` walks them in that order, returning the first
11//! non-empty list.
12
13use anyhow::Result;

84/// Probe the catalog for `family` without constructing a live provider.
85/// Returns the union of every known public catalog source. Errors if
86/// `family` is unknown or has no public catalog source set.
87pub async fn list_models_for_family(family: &str) -> Result<Vec<String>> {
88    let Some((md_key, or_prefix)) = catalog_source_for(family) else {
89        anyhow::bail!("unknown provider family {family:?}");
90    };
91    if let Some(k) = md_key
92        && let Ok(ms) = crate::models_dev::list_models_for(k).await
93        && !ms.is_empty()
94    {
95        return Ok(ms);
96    }
97    if let Some(p) = or_prefix {
98        return crate::openrouter_catalog::list_models_for_vendor(p).await;
99    }
100    anyhow::bail!("no public catalog for family {family:?}")
101}

103#[cfg(test)]
104mod tests {
105    use super::*;
106
107    /// `catalog_source_for` must classify every canonical family the
108    /// `for_each_model_provider_slot!` macro emits. Drift catches a new
109    /// slot added to the macro without a matching catalog-table entry —
110    /// `catalog_source_for` would return `None` and the gateway endpoint
111    /// would surface `unknown provider family` for that family.
112    #[test]
113    fn every_canonical_family_has_a_catalog_table_entry() {
114        macro_rules! collect_family_names {
115            ($(($field:ident, $type_str:literal, $cfg_ty:ty)),+ $(,)?) => {
116                vec![$($type_str),+]
117            };
118        }
119        let families: Vec<&str> =
120            zeroclaw_config::for_each_model_provider_slot!(collect_family_names);
121        let mut missing: Vec<&str> = Vec::new();
122        for family in &families {
123            if catalog_source_for(family).is_none() {
124                missing.push(family);
125            }
126        }
127        assert!(
128            missing.is_empty(),
129            "catalog_source_for is missing entries for: {missing:?}"
130        );
131    }
132
133    #[test]
134    fn unknown_family_returns_none() {
135        assert!(catalog_source_for("not_a_real_provider").is_none());
136    }
137
138    #[test]
139    fn known_family_with_dual_sources_returns_both() {
140        let (md, or) = catalog_source_for("xai").expect("xai is canonical");
141        assert_eq!(md, Some("xai"));
142        assert_eq!(or, Some("x-ai"));
143    }
144
145    #[test]
146    fn local_only_family_returns_no_sources() {
147        let (md, or) = catalog_source_for("llamacpp").expect("llamacpp is canonical");
148        assert_eq!(md, None);
149        assert_eq!(or, None);
150    }
151
152    #[test]
153    fn bespoke_family_with_only_models_dev() {
154        let (md, or) = catalog_source_for("azure").expect("azure is canonical");
155        assert_eq!(md, Some("azure"));
156        assert_eq!(or, None);
157    }
158
159    #[test]
160    fn bespoke_family_with_only_openrouter() {
161        let (md, or) = catalog_source_for("ai21").expect("ai21 is canonical");
162        assert_eq!(md, None);
163        assert_eq!(or, Some("ai21"));
164    }
