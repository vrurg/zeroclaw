1//! Runtime-spawned ephemeral sub-agents that inherit their parent
2//! agent's identity by default: same UUID, same `SecurityPolicy`, same
3//! memory allowlist. A SubAgent run is auditable as a child of the
4//! parent and stays inside the parent's permissions envelope.
5//!
6//! Two spawn sites converge on [`SubAgentSpawn`]: the agent-loop tool
7//! `spawn_subagent` and the cron scheduler's `JobType::Agent` dispatch.
8//! Sharing the surface keeps permission inheritance, tracing-span
9//! shape, and audit attribution uniform.
10//!
11//! Power-users may narrow a SubAgent's permissions via
12//! [`SubAgentOverrides`]; [`SubAgentSpawn::build`] validates each
13//! override as a subset of the parent (using
14//! [`SecurityPolicy::ensure_no_escalation_beyond`] for the policy and
15//! an alias-set containment check for the memory allowlist) and
16//! returns `Err` with the originating violation chained on any
17//! escalation.
18//!
19//! The memory allowlist is carried as a set of agent **aliases** (the
20//! `[agents.<alias>]` config keys), not backend storage identifiers.
21//! Consumers that build an [`zeroclaw_memory::AgentScopedMemory`] must resolve aliases
22//! to backend identifiers via
23//! [`zeroclaw_memory::Memory::ensure_agent_uuid`] first — SQL-backed
24//! stores use UUIDs from the `agents` table; Markdown / Qdrant / None
25//! use the alias verbatim per the trait default. Holding aliases at
26//! this layer means [`SubAgentSpawn::for_agent`] does not need a
27//! backend handle to construct.
28

33use zeroclaw_config::policy::SecurityPolicy;
34use zeroclaw_config::schema::Config;

36/// Optional narrowing applied to a SubAgent at spawn time. `None` on
37/// every field means "inherit parent verbatim"; `Some(...)` narrows.
38/// Each field is independently validated by [`SubAgentSpawn::build`]
39/// to reject any value that escalates beyond the parent.
40///
41/// The default-everything-inherits model means the common case is
42/// `SubAgentOverrides::default()` — a no-op.
43#[derive(Debug, Clone, Default)]
44pub struct SubAgentOverrides {
45    /// Override the SubAgent's [`SecurityPolicy`]. Validated as a
46    /// subset of the parent via
47    /// [`SecurityPolicy::ensure_no_escalation_beyond`].
48    pub policy: Option<SecurityPolicy>,
49    /// Override the SubAgent's memory allowlist (the set of sibling
50    /// agent **aliases** the SubAgent may recall from, as written in
51    /// `[agents.<alias>]` keys). Validated as a subset of the
52    /// parent's allowlist; any alias here that is not on the parent's
53    /// list is rejected.
54    ///
55    /// These are config-layer aliases, not backend storage
56    /// identifiers. Consumers that build an [`zeroclaw_memory::AgentScopedMemory`]
57    /// must resolve aliases to backend identifiers via
58    /// [`zeroclaw_memory::Memory::ensure_agent_uuid`] before passing
59    /// them to the wrapper (SQL backends use UUIDs; Markdown / Qdrant
60    /// / None use the alias verbatim per the trait default). The
61    /// in-tree consumer today is `zeroclaw_memory::create_memory_for_agent`,
62    /// which performs the resolution.
63    pub allowed_agent_aliases: Option<HashSet<String>>,
64}
65
66/// Constructed SubAgent context: bound parent identity, validated
67/// child policy, and the resolved memory allowlist.
68#[derive(Debug, Clone)]
69pub struct SubAgentContext {
70    /// The parent agent's alias (e.g. `"researcher"`). SubAgents share
71    /// the parent's identity at the data layer (no separate row in the
72    /// `agents` table); the distinction between parent and sub-run is
73    /// captured at the tracing-span level
74    /// (`agent.<alias>.subagent.<run_id>`).
75    pub parent_alias: String,
76    /// The validated [`SecurityPolicy`] this SubAgent operates under.
77    /// Identical to the parent's when `SubAgentOverrides::policy` is
78    /// `None`; otherwise a narrowed copy that passed
79    /// [`SecurityPolicy::ensure_no_escalation_beyond`].
80    pub policy: Arc<SecurityPolicy>,
81    /// Resolved memory allowlist as a set of agent **aliases**. The
82    /// bound `parent_alias` is always included so the SubAgent always
83    /// sees the parent's own rows; the rest is either the parent's
84    /// allowlist verbatim or a validated subset.
85    ///
86    /// See [`SubAgentOverrides::allowed_agent_aliases`] for the
87    /// alias-vs-backend-identifier distinction; consumers that build
88    /// an [`zeroclaw_memory::AgentScopedMemory`] must resolve to backend identifiers
89    /// before passing the set to the wrapper.
90    pub allowed_agent_aliases: HashSet<String>,
91}

93/// Builder for a SubAgent spawn. The caller resolves a parent agent
94/// from the loaded config; [`Self::build`] applies any narrowing
95/// overrides and validates the result.
96#[derive(Debug)]
97pub struct SubAgentSpawn {
98    pub parent_alias: String,
99    pub parent_policy: Arc<SecurityPolicy>,
100    pub parent_allowed_agent_aliases: HashSet<String>,
101}
102
103impl SubAgentSpawn {
104    /// Resolve a parent's identity from the loaded config and an
105    /// agent alias. Returns `Err` when the alias does not name a
106    /// configured agent — the spawn site surfaces a structured
107    /// failure instead of invoking the agent loop on a nonexistent
108    /// identity.
109    ///
110    /// The parent policy is rebuilt from config via
111    /// [`SecurityPolicy::for_agent`]. This is the right entry point
112    /// for spawn sites with **no live parent context** — most
113    /// importantly the cron scheduler's `JobType::Agent` dispatch,
114    /// which has no session and must use the per-agent install
115    /// workspace as the sandbox boundary.
116    ///
117    /// Interactive spawn sites that hold the parent's live
118    /// `Arc<SecurityPolicy>` (the agent-loop `spawn_subagent` tool and
119    /// the `delegate` tool when called from an ACP/gateway session)
120    /// must use [`Self::for_agent_with_policy`] instead, so that
121    /// session-scoped policy fields — most importantly
122    /// `workspace_dir`, which IDE/ACP clients pin to the session cwd
123    /// — survive the spawn. See issue #7263.
124    pub fn for_agent(config: &Config, agent_alias: &str) -> Result<Self> {
125        // Upfront alias check so a missing-agent failure surfaces with
126        // the "no agent configured under alias …" message rather than
127        // the policy resolver's less specific "no resolvable
128        // risk_profile" wrapping.
129        if !config.agents.contains_key(agent_alias) {
130            anyhow::bail!("no agent configured under alias {agent_alias:?}");
131        }
132        let parent_policy = SecurityPolicy::for_agent(config, agent_alias)
133            .map(Arc::new)
134            .with_context(|| {
135                format!("could not resolve security policy for agent {agent_alias:?}")
136            })?;
137        Self::for_agent_with_policy(config, agent_alias, parent_policy)
138    }
139
140    /// Resolve a parent's identity using a **pre-built** security
141    /// policy — the live `Arc<SecurityPolicy>` that the parent's tool
142    /// registry is using. This is the spawn path interactive sites
143    /// (ACP `spawn_subagent`, ACP `delegate`) must take so that
144    /// session-scoped policy fields — most importantly
145    /// `workspace_dir`, which IDE/ACP clients pin to the session cwd
146    /// — survive the spawn. Without this hook the policy is rebuilt
147    /// from config and the session override is silently dropped
148    /// (issue #7263).
149    ///
150    /// The memory allowlist is still resolved from config because it
151    /// is declared statically per agent and the policy carries no
152    /// equivalent field.
153    pub fn for_agent_with_policy(
154        config: &Config,
155        agent_alias: &str,
156        parent_policy: Arc<SecurityPolicy>,
157    ) -> Result<Self> {
158        let agent = config
159            .agents
160            .get(agent_alias)
161            .with_context(|| format!("no agent configured under alias {agent_alias:?}"))?;
162
163        let mut parent_allowed_agent_aliases: HashSet<String> = agent
164            .workspace
165            .read_memory_from
166            .iter()
167            .map(|alias| alias.as_str().to_string())
168            .collect();
169        parent_allowed_agent_aliases.insert(agent_alias.to_string());
170
171        Ok(Self {
172            parent_alias: agent_alias.to_string(),
173            parent_policy,
174            parent_allowed_agent_aliases,
175        })
176    }
177
178    /// Apply `overrides` to the parent's permissions and return a
179    /// validated [`SubAgentContext`]. On any escalation, returns
180    /// `Err` with the originating violation in the error chain.
181    ///
182    /// When the caller supplies a policy override, the child inherits
183    /// the parent's `PerSenderTracker` so action and cost budgets are
184    /// shared between parent and SubAgent runs. Otherwise a SubAgent
185    /// could be spawned to bypass the parent's `max_actions_per_hour`
186    /// or `max_cost_per_day_cents` ceiling by consuming from a
187    /// fresh-zeroed bucket; the inheritance closes that escape. The
188    /// no-override path already shares the bucket via
189    /// `Arc<SecurityPolicy>` cloning.
190    pub fn build(self, overrides: SubAgentOverrides) -> Result<SubAgentContext> {
191        let policy = if let Some(mut child_policy) = overrides.policy {
192            child_policy
193                .ensure_no_escalation_beyond(&self.parent_policy)
194                .map_err(|violation| {
195                    ::zeroclaw_log::record!(
196                        WARN,
197                        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
198                            .with_outcome(::zeroclaw_log::EventOutcome::Failure)
199                            .with_attrs(::serde_json::json!({
200                                "violation": violation.to_string(),
201                            })),
202                        "subagent build refused: policy override escalates beyond parent"
203                    );
204                    anyhow::Error::msg(format!(
205                        "subagent policy override escalates beyond parent: {violation}"
206                    ))
207                })?;
208            // Share the parent's action/cost tracker. `PerSenderTracker`
209            // is `Clone` (deep-copy of buckets) but the SubAgent must
210            // see the parent's live bucket state, not a frozen
211            // snapshot, so steal the parent's tracker by cloning the
212            // inner `Arc<SecurityPolicy>` once and assigning the
213            // child's `tracker` field from it.
214            child_policy.tracker = self.parent_policy.tracker.clone();
215            Arc::new(child_policy)
216        } else {
217            self.parent_policy.clone()
218        };
219
220        let allowed_agent_aliases = if let Some(child_allowed) = overrides.allowed_agent_aliases {
221            for alias in &child_allowed {
222                if !self.parent_allowed_agent_aliases.contains(alias) {
223                    anyhow::bail!(
224                        "subagent allowlist override contains alias {alias:?} not present on \
225                         parent's memory allowlist; SubAgent overrides may only narrow"
226                    );
227                }
228            }
229            let mut resolved = child_allowed;
230            resolved.insert(self.parent_alias.clone());
231            resolved
232        } else {
233            self.parent_allowed_agent_aliases
234        };
235
236        Ok(SubAgentContext {
237            parent_alias: self.parent_alias,
238            policy,
239            allowed_agent_aliases,
240        })
241    }
242}

250    fn config_with_agent(alias: &str) -> Config {
251        let mut config = Config::default();
252        config
253            .risk_profiles
254            .insert("default".to_string(), RiskProfileConfig::default());
255        config.agents.insert(
256            alias.to_string(),
257            AliasedAgentConfig {
258                risk_profile: "default".to_string(),
259                ..AliasedAgentConfig::default()
260            },
261        );
262        config
263    }

265    #[test]
266    fn for_agent_resolves_parent_identity_from_config() {
267        let config = config_with_agent("alpha");
268        let ctx = SubAgentSpawn::for_agent(&config, "alpha")
269            .expect("for_agent must succeed for a configured agent")
270            .build(SubAgentOverrides::default())
271            .expect("inherits-verbatim build must succeed");
272        assert_eq!(ctx.parent_alias, "alpha");
273        assert!(
274            ctx.allowed_agent_aliases.contains("alpha"),
275            "an agent always sees its own rows"
276        );
277    }

358    #[test]
359    fn build_with_override_inherits_parent_action_budget() {
360        // SubAgent runs must consume from the parent's action budget
361        // so spawning children cannot bypass `max_actions_per_hour`.
362        // The override path (caller-supplied policy) is the one with
363        // the bug; the inherit-verbatim path is correct by Arc reuse.
364        let config = config_with_agent("alpha");
365        let spawn = SubAgentSpawn::for_agent(&config, "alpha").unwrap();
366        let parent_policy = spawn.parent_policy.clone();
367
368        // Burn the parent's action budget right up to the ceiling so
369        // the child's first record_action would push past it.
370        for _ in 0..parent_policy.max_actions_per_hour {
371            assert!(
372                parent_policy.record_action(),
373                "parent budget should accept records up to its ceiling"
374            );
375        }
376
377        // Build a child policy that's a subset of the parent (no
378        // escalation) but with the default fresh tracker. The fix
379        // copies the parent's tracker into the child so the next
380        // record_action sees the parent's exhausted bucket.
381        let child_policy = (*parent_policy).clone();
382        let ctx = spawn
383            .build(SubAgentOverrides {
384                policy: Some(child_policy),
385                ..SubAgentOverrides::default()
386            })
387            .expect("inheriting policy as a subset must succeed");
388
389        assert!(
390            !ctx.policy.record_action(),
391            "child must inherit parent's exhausted action budget; \
392             a fresh bucket here means the budget is bypass-able by \
393             spawning a SubAgent"
394        );
395    }

397    /// Regression for issue #7263: when an interactive spawn site
398    /// (ACP `spawn_subagent` / `delegate`) supplies a pre-built parent
399    /// policy whose `workspace_dir` was pinned to the session cwd,
400    /// `for_agent_with_policy` must propagate that policy verbatim
401    /// instead of regenerating one from config. Without this the
402    /// child's file/shell tools jail to `~/.zeroclaw/agents/<alias>/
403    /// workspace` rather than the IDE's session cwd, breaking
404    /// subagent-driven workflows in repos outside the install root.
405    #[test]
406    fn for_agent_with_policy_preserves_session_workspace_dir() {
407        let config = config_with_agent("alpha");
408
409        // The session cwd is some directory that is NOT
410        // `config.agent_workspace_dir("alpha")`. Pick an absolute path
411        // that's stable across hosts.
412        let session_cwd = PathBuf::from("/tmp/zeroclaw-test-session-cwd-7263");
413        let config_workspace = config.agent_workspace_dir("alpha");
414        assert_ne!(
415            session_cwd, config_workspace,
416            "test precondition: session cwd must differ from config workspace"
417        );
418
419        // Build the "live" parent policy the way the interactive
420        // builders do (config-derived, then session_cwd override).
421        let mut live_policy = SecurityPolicy::for_agent(&config, "alpha").unwrap();
422        live_policy.workspace_dir = session_cwd.clone();
423        let live_policy = Arc::new(live_policy);
424
425        let ctx = SubAgentSpawn::for_agent_with_policy(&config, "alpha", live_policy.clone())
426            .expect("for_agent_with_policy must accept a live parent policy")
427            .build(SubAgentOverrides::default())
428            .expect("inherits-verbatim build must succeed");
429
430        // The child policy must be the same Arc (no clone, no rebuild)
431        // and must carry the session cwd through to the loop.
432        assert!(
433            Arc::ptr_eq(&ctx.policy, &live_policy),
434            "default overrides must reuse the parent's Arc, not regenerate"
435        );
436        assert_eq!(
437            ctx.policy.workspace_dir, session_cwd,
438            "session cwd must survive the spawn; regression for issue #7263"
439        );
440    }
