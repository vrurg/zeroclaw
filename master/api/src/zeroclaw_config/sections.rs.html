1//! Curated sections surface — a flat ordered set of [`Section`]s the
2//! operator walks (new install) or scans (returning user) to configure
3//! a working ZeroClaw deployment.
4//!
5//! Every fact about a section (its enum variant, its on-the-wire key,
6//! its UI shape, its help blurb, its canonical position) lives in ONE
7//! table — the `sections!` invocation below. The macro expands that
8//! table into the [`Section`] enum, every per-variant `match` helper,
9//! and the [`QUICKSTART_SECTIONS`] const, so adding a section is exactly
10//! one row, no hand-listed variant set anywhere else.
11//!
12//! Consumers (CLI runtime, gateway, dashboard) dispatch off this enum;
13//! drift is a compile error.
14
15use serde::{Deserialize, Serialize};

17/// UI rendering shape for a section. Drives picker / form dispatch on
18/// the `/config` curated section explorer and the Quickstart flow.
19#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
20#[cfg_attr(feature = "schema-export", derive(schemars::JsonSchema))]
21#[serde(rename_all = "snake_case")]
22pub enum SectionShape {
23    /// `<section>` renders a schema-driven form with no picker step.
24    DirectForm,
25    /// `<section>.<alias>` map of structured entries; the section page
26    /// shows an alias list with `+ Add` and clicking an alias opens its
27    /// schema form.
28    OneTierAliasMap,
29    /// `<section>.<type>.<alias>` two-tier map. Picker chooses `<type>`,
30    /// alias-list step chooses `<alias>`, then the schema form opens.
31    TypedFamilyMap,
32    /// Single non-alias choice (memory backend, tunnel provider). Picker
33    /// flips a top-level field, then the schema form for the chosen
34    /// backend/provider renders.
35    BackendPicker,
36}
37
38/// Single source of truth for every pickable config section. Each row
39/// maps 1:1 to a dashboard `/config/<key>` page, a CLI
40/// `zeroclaw quickstart` flow and the gateway section picker handler.
41/// Adding/removing a section is one row here and every consumer's
42/// `match` either compiles cleanly or fails with an exhaustiveness
43/// error pointing at exactly what needs an arm.
44///
45/// Row order is the canonical order operators see in the dashboard
46/// and walk through in the CLI. It is dependency-correct: every
47/// downstream alias reference an Agent carries (model_provider,
48/// risk_profile, runtime_profile, channels, *_bundles) appears earlier
49/// in the list than [`Section::Agents`], so walking top-to-bottom
50/// never produces a dangling reference.
51macro_rules! sections {
52    (
53        $(
54            $var:ident => {
55                key:   $key:literal,
56                shape: $shape:ident,
57                help:  $help:expr $(,)?
58            }
59        ),+ $(,)?
60    ) => {
61        /// One pickable section. The variant ordering follows the
62        /// `sections!` macro invocation.
63        ///
64        /// With the `clap` feature on, this enum doubles as the
65        /// `zeroclaw quickstart` and curated-section endpoints — no separate
66        /// mirror enum in the binary crate.
67        #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
68        #[cfg_attr(feature = "schema-export", derive(schemars::JsonSchema))]
69        #[cfg_attr(feature = "clap", derive(clap::Subcommand))]
70        #[serde(rename_all = "snake_case")]
71        pub enum Section {
72            $(
73                // Both clap (`--help`) and our runtime `help()` method
74                // need the same blurb; emit it once as a doc comment so
75                // the two surfaces share a single string per variant.
76                #[doc = $help]
77                #[cfg_attr(feature = "clap", command(name = $key))]
78                $var,
79            )+
80        }
81
82        impl Section {
83            /// Stable on-the-wire key. Also serves as the TOML
84            /// top-level prefix (e.g. `providers.models.<type>.<alias>`),
85            /// the curated section URL segment, and the
86            /// `SectionInfo.key` field returned by the gateway.
87            #[must_use]
88            pub const fn as_str(self) -> &'static str {
89                match self {
90                    $( Self::$var => $key, )+
91                }
92            }
93
94            /// Editor shape — the dashboard and the CLI both
95            /// dispatch off this so the same component lights up for
96            /// the same section in both surfaces.
97            #[must_use]
98            pub const fn shape(self) -> SectionShape {
99                match self {
100                    $( Self::$var => SectionShape::$shape, )+
101                }
102            }
103
104            /// Per-section help blurb — single source of truth for
105            /// the copy shown above the section's picker / form on
106            /// every surface (CLI `ui.note(...)`, TUI heading,
107            /// dashboard `SectionInfo.help`).
108            #[must_use]
109            pub const fn help(self) -> &'static str {
110                match self {
111                    $( Self::$var => $help, )+
112                }
113            }
114
115            /// Parse a stable wire key, tolerating both the snake and
116            /// kebab spellings of any section. The schema mixes the two:
117            /// `model_providers` (snake) and `peer-groups` (kebab) are
118            /// both valid wire forms produced elsewhere in the codebase.
119            /// Callers (dashboard URL routing, gateway picker dispatch,
120            /// CLI clap subcommands) can pass either form; `from_key`
121            /// resolves to the same variant. Returns `None` for keys
122            /// outside the known section table. Named `from_key` rather
123            /// than `from_str` so clippy doesn't flag it as confusable
124            /// with `std::str::FromStr` (parse failure is `None`, not
125            /// `Err(_)`).
126            #[must_use]
127            pub fn from_key(s: &str) -> Option<Self> {
128                let try_match = |s: &str| -> Option<Self> {
129                    match s {
130                        $( $key => Some(Self::$var), )+
131                        _ => None,
132                    }
133                };
134                if let Some(v) = try_match(s) {
135                    return Some(v);
136                }
137                if s.contains('_')
138                    && let Some(v) = try_match(&s.replace('_', "-"))
139                {
140                    return Some(v);
141                }
142                if s.contains('-')
143                    && let Some(v) = try_match(&s.replace('-', "_"))
144                {
145                    return Some(v);
146                }
147                None
148            }
149        }
150
151        /// Canonical ordering of sections enumerated by
152        /// the Quickstart flow and the curated section explorer. The
153        /// dashboard renders Next/Finish navigation against this list.
154        /// Every consumer that needs section ordering reads from here.
155        pub const QUICKSTART_SECTIONS: &[Section] = &[ $( Section::$var ),+ ];
156    };
157}

329/// Canonical-order index of `section` in [`QUICKSTART_SECTIONS`].
330/// Always `Some` for any valid `Section` variant — the const includes
331/// every variant by construction. Returns `Option` for API symmetry
332/// with [`section_index_for_key`], which can fail on unknown keys.
333#[must_use]
334pub fn section_index(section: Section) -> Option<usize> {
335    QUICKSTART_SECTIONS.iter().position(|s| *s == section)
336}
337
338/// Canonical-order index for a wire key, or `None` if the key isn't a
339/// known [`Section`]. Used by gateway / dashboard sort comparators that
340/// take string keys from the HTTP layer.
341#[must_use]
342pub fn section_index_for_key(key: &str) -> Option<usize> {
343    Section::from_key(key).and_then(section_index)
344}

347#[must_use]
348pub fn is_known_section(key: &str) -> bool {

352/// Help blurb for a section key, covering both `Section` variants and
353/// the long tail of top-level `Config` fields the dashboard / TUI config
354/// editor surface (gateway, scheduler, observability, …). Single source
355/// of truth shared by every surface — the gateway sidebar, the CLI
356/// Quickstart flow, and the future TUI config editor all call this rather
357/// than maintaining parallel tables.
358///
359/// Resolution order:
360/// 1. `Section` variants (curated `help` text next to the variant
361///    declaration in the `sections!` macro).
362/// 2. The `Config` struct's `#[nested]` field-level `///` docstring,
363///    harvested by the `Configurable` derive into
364///    `Config::nested_section_help`. This is what makes adding a new
365///    top-level section a one-line schema change with no parallel
366///    help table to update.
367///
368/// Returns `""` for keys without a docstring so callers can decide
369/// whether to omit the help row or show a fallback.
370#[must_use]
371pub fn section_help(key: &str) -> &'static str {
372    if let Some(s) = Section::from_key(key) {
373        return s.help();
374    }
375    crate::schema::Config::nested_section_help(key).unwrap_or("")
376}

378/// First segment of a dotted property path mapped back to the section
379/// it lives under, or `None` for non-section paths
380/// (`onboard_state.completed_sections`, etc.).
381#[must_use]
382pub fn section_for_path(path: &str) -> Option<Section> {
383    Section::from_key(path.split('.').next()?)
384}

386/// Does this section show any signal of having been touched on this
387/// install? Used by callers (RPC config-list filtering, lifecycle
388/// dispatch) to decide whether to surface a section as "untouched".
389///
390/// Each variant decides what counts as a real signal vs a default
391/// value that round-trips identically across a fresh install.
392pub fn section_has_signal(cfg: &crate::schema::Config, section: Section) -> bool {
393    match section {
394        Section::ModelProviders => !cfg.providers.models.is_empty(),
395        // `channels.cli: bool` is a default-true scalar that lives directly
396        // under `channels.*`, so a bare `starts_with("channels.")` check
397        // fires on every fresh install. Require a nested channel config
398        // (e.g. `channels.telegram.bot-token`) — anything with a second dot
399        // segment — to count as user-driven signal.
400        Section::Channels => cfg.prop_fields().iter().any(|f| {
401            f.name
402                .strip_prefix("channels.")
403                .is_some_and(|rest| rest.contains('.'))
404        }),
405        Section::Hardware => cfg.hardware.enabled,
406        // Memory's default backend is "sqlite" and Tunnel's is "none" —
407        // both are valid user choices indistinguishable from untouched
408        // defaults. TTS / transcription providers and agents start
409        // empty; their existence in the typed family map IS the signal,
410        // not a derivable default-divergence. Marker-only for these.
411        Section::TtsProviders
412        | Section::TranscriptionProviders
413        | Section::Memory
414        | Section::Tunnel
415        | Section::Agents
416        | Section::Skills
417        | Section::SkillBundles
418        | Section::RiskProfiles
419        | Section::RuntimeProfiles
420        | Section::PeerGroups
421        | Section::Storage
422        | Section::Cron
423        | Section::Mcp
424        | Section::McpBundles
425        | Section::KnowledgeBundles
426        | Section::QuickstartState => false,
427    }
428}

434    /// Round-trip every entry in the canonical list. `from_key`,
435    /// `as_str`, `section_index`, and `QUICKSTART_SECTIONS` are all
436    /// generated from the same `sections!` row, so this test exercises
437    /// the table — adding a row that breaks any of them fails here
438    /// without listing variants by hand.
439    #[test]
440    fn sections_round_trip() {
441        for s in QUICKSTART_SECTIONS {
442            assert_eq!(Section::from_key(s.as_str()), Some(*s), "{s} round-trip");
443            assert_eq!(
444                section_index(*s),
445                Some(QUICKSTART_SECTIONS.iter().position(|x| x == s).unwrap()),
446            );
447        }
448        assert_eq!(Section::from_key("gateway"), None);
449        assert_eq!(Section::from_key("not_a_section"), None);
450    }

452    /// Every section the dashboard URL surface points at must resolve
453    /// through `Section::from_key`. The dashboard URL form is kebab-case
454    /// (`peer-groups`), the canonical wire form may be snake_case
455    /// (`peer_groups`); both must parse to the same variant.
456    #[test]
457    fn dashboard_url_sections_round_trip_kebab_and_snake() {
458        let kebab_then_snake: &[(&str, &str, Section)] = &[
459            ("peer-groups", "peer_groups", Section::PeerGroups),
460            ("mcp-bundles", "mcp_bundles", Section::McpBundles),
461            (
462                "knowledge-bundles",
463                "knowledge_bundles",
464                Section::KnowledgeBundles,
465            ),
466            ("skill-bundles", "skill_bundles", Section::SkillBundles),
467            ("risk-profiles", "risk_profiles", Section::RiskProfiles),
468            (
469                "runtime-profiles",
470                "runtime_profiles",
471                Section::RuntimeProfiles,
472            ),
473            ("storage", "storage", Section::Storage),
474            ("cron", "cron", Section::Cron),
475            ("mcp", "mcp", Section::Mcp),
476        ];
477        for (kebab, snake, expected) in kebab_then_snake {
478            assert_eq!(
479                Section::from_key(kebab),
480                Some(*expected),
481                "kebab `{kebab}` should resolve to {expected:?}",
482            );
483            assert_eq!(
484                Section::from_key(snake),
485                Some(*expected),
486                "snake `{snake}` should resolve to {expected:?}",
487            );
488            assert!(
489                QUICKSTART_SECTIONS.contains(expected),
490                "{expected:?} must be in QUICKSTART_SECTIONS",
491            );
492        }
493    }

495    /// Every OneTierAliasMap section's wire key must appear verbatim
496    /// in `Config::map_key_sections()`. That table is what
497    /// `Config::create_map_key` dispatches off, so a mismatch silently
498    /// breaks the dashboard's `+ Add` affordance.
499    #[test]
500    fn alias_map_section_wire_keys_match_map_key_sections() {
501        use crate::schema::Config;
502        let sections = Config::map_key_sections();
503        let paths: std::collections::BTreeSet<&str> = sections.iter().map(|s| s.path).collect();
504        let alias_map_sections = [
505            Section::PeerGroups,
506            Section::Cron,
507            Section::McpBundles,
508            Section::KnowledgeBundles,
509            Section::SkillBundles,
510            Section::RiskProfiles,
511            Section::RuntimeProfiles,
512        ];
513        for section in alias_map_sections {
514            assert!(
515                paths.contains(section.as_str()),
516                "`Section::{section:?}.as_str() = {}` is not in map_key_sections; the \
517                 picker's create_map_key call site will fail. Registered paths: {paths:?}",
518                section.as_str(),
519            );
520        }
521    }

523    /// Canonical order is dependency-correct: every Section that
524    /// `AliasedAgentConfig` references through an alias field appears
525    /// earlier in the list than `Section::Agents`. Walking
526    /// `QUICKSTART_SECTIONS` top-to-bottom never asks the operator to
527    /// configure an Agent before the things it has to bind to exist.
528    #[test]
529    fn ordering_respects_agent_dependency_tiers() {
530        let idx = |s: Section| {
531            QUICKSTART_SECTIONS
532                .iter()
533                .position(|x| *x == s)
534                .unwrap_or_else(|| panic!("{s:?} missing from QUICKSTART_SECTIONS"))
535        };
536
537        // Brain + behavior shape + bundles + channels all precede Agents.
538        for upstream in [
539            Section::ModelProviders,
540            Section::RiskProfiles,
541            Section::RuntimeProfiles,
542            Section::SkillBundles,
543            Section::McpBundles,
544            Section::KnowledgeBundles,
545            Section::Channels,
546        ] {
547            assert!(
548                idx(upstream) < idx(Section::Agents),
549                "{upstream:?} must precede Agents (Agent references it through an alias field)",
550            );
551        }
552
553        // Storage precedes Memory (memory.backend = "<storage_type>.<alias>").
554        assert!(
555            idx(Section::Storage) < idx(Section::Memory),
556            "Storage must precede Memory (memory.backend points at a storage instance)",
557        );
558
559        // Topology references agents.
560        for downstream in [Section::PeerGroups, Section::Cron] {
561            assert!(
562                idx(Section::Agents) < idx(downstream),
563                "{downstream:?} references agents and must follow Agents in the canonical order",
564            );
565        }
566    }

568    /// Storage help must steer first-time operators toward SQLite as the
569    /// safe default. Pins the contract: SQLite is named, flagged as a
570    /// default/safe/recommended choice, and positioned before the
571    /// alternatives so the recommendation lands first instead of being
572    /// buried in a closing list.
573    #[test]
574    fn storage_help_steers_to_sqlite_default() {
575        let help = section_help("storage").to_lowercase();
576        let sqlite_pos = help
577            .find("sqlite")
578            .expect("storage help must mention SQLite by name");
579        assert!(
580            help.contains("default") || help.contains("safe") || help.contains("recommend"),
581            "storage help must signal SQLite is the default/safe/recommended choice; got: {help}",
582        );
583        for other in ["postgres", "qdrant", "markdown", "lucid"] {
584            let other_pos = help.find(other).unwrap_or_else(|| {
585                panic!(
586                    "storage help must still name `{other}` so operators know the alternatives \
587                     exist; got: {help}",
588                )
589            });
590            assert!(
591                sqlite_pos < other_pos,
592                "SQLite (at {sqlite_pos}) must be mentioned before `{other}` (at {other_pos}) so \
593                 the default recommendation lands first",
594            );
595        }
596    }
